  def initialize(symbol, entity_class, options, &block)
    @symbol, @entity_class, @block = symbol, entity_class, block
    @options = {
      :ignores => []
    }.merge(options || {})
    @options[:ignores] = [@options[:ignores]] unless @options[:ignores].respond_to?(:include?)
  end
